    Subsection Problem 1

    \begin{align*} \binom{n}{k} \binom{k}{r}\amp=\frac{n!}{k!(n-k)!}\cdot \frac{k!}{r!(k-r)!}\\ \amp=\frac{n!}{r!}\cdot \frac{1}{(n-k)!}\cdot \frac{1}{(k-r)!}\\ \amp= \frac{n!}{r!(n-r)!}\cdot \frac{1}{(n-k)!}\cdot \frac{(n-r)!}{(k-r)!}\\ \amp= \frac{n!}{r!(n-r)!}\cdot \frac{1}{((n-r)-(k-r))!}\cdot \frac{(n-r)!}{(k-r)!}\\ \amp=\binom{n}{r}\binom{n-r}{k-r} \end{align*}
    Let \(r=1\) in the identity:
    \begin{align*} \binom{n}{k}\binom{k}{1}\amp=\binom{n}{1}\binom{n-1}{k-1}\\ \binom{n}{k}\amp=\frac{n}{k}\cdot \frac{(n-1)!}{(k-1)!((n-1)-(k-1))!}\\ \amp=\frac{n!}{k\cdot (k-1)!\cdot (n-k)!}\\ \amp=\frac{n-k+1}{k}\cdot \frac{n!}{(k-1)!(n-k)!(n-k+1)}\\ \amp= \frac{n-k+1}{k}\cdot \frac{n!}{(k-1)!((n-k+1)!)}\\ \amp= \frac{n-k+1}{k}\cdot\binom{n}{k-1} \end{align*}

    Subsection Problem 4

    \begin{equation*} \binom{n}{r}=\frac{n!}{r!(n-r)!} \end{equation*}
    \begin{equation*} \binom{n}{r+1}=\frac{n!}{(r+1)!(n-r-1)!} \end{equation*}
    So
    \begin{align*} \binom{n}{r} <\binom{n}{r+1} \amp \phantom{4+5} \text{ iff } \phantom{4+5} r!(n-r)!>(r+1)!(n-r-1)!\\ \amp \phantom{4+5} \text{ iff } \phantom{4+5} n-r >r+1\\ \amp \phantom{4+5} \text{ iff } \phantom{4+5} n-1 >2r\\ \amp \phantom{4+5} \text{ iff } \phantom{4+5} \frac{1}{2}(n-1) > r. \end{align*}
    Of course, we need \(r \geq 0\text{.}\) The other parts are similar.
